package org.example;

import java.awt.*;
import java.awt.event.*;
import java.io.*;
import java.util.*;

public class Web_res_lab extends Frame implements ActionListener{

    Button exitButton = new Button("Exit");
    Button searchButton = new Button("Search");
    TextArea textArea = new TextArea();

    public Web_res_lab() {
        super("Dog Breed Search");
        setLayout(null);
        setBackground(new Color(150, 200, 100));
        setSize(450, 250);

        exitButton.setBounds(110, 190, 100, 20);
        exitButton.addActionListener(this);
        add(exitButton);

        searchButton.setBounds(110, 165, 100, 20);
        searchButton.addActionListener(this);
        add(searchButton);

        textArea.setBounds(20, 50, 300, 100);
        add(textArea);

        setLocationRelativeTo(null);
        setVisible(true);

        addWindowListener(new WindowAdapter() {
            @Override
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }

    @Override
    public void actionPerformed(ActionEvent ae) {
        if (ae.getSource() == exitButton) {
            System.exit(0);
        } else if (ae.getSource() == searchButton) {
            String input = textArea.getText().trim();
            if (input.isEmpty()) {
                textArea.setText("Please enter dog qualities (comma-separated)");
                return;
            }
            
            String[] qualities = input.split(",");
            for (int i = 0; i < qualities.length; i++) {
                qualities[i] = qualities[i].trim().toLowerCase();
            }
            
            textArea.setText("Searching for breeds with qualities: " + Arrays.toString(qualities) + "\n\n");
            
            try {
                File breedsFile = new File("dog_breeds.txt");
                if (!breedsFile.exists()) {
                    textArea.append("Error: dog_breeds.txt file not found!");
                    return;
                }
                
                List<String> matchingBreeds = new ArrayList<>();
                try (BufferedReader reader = new BufferedReader(new FileReader(breedsFile))) {
                    String line;
                    while ((line = reader.readLine()) != null) {
                        if (line.contains(":")) {
                            String[] parts = line.split(":", 2);
                            String breedName = parts[0].trim();
                            String breedQualities = parts[1].toLowerCase();
                            
                            int matchCount = 0;
                            for (String quality : qualities) {
                                if (breedQualities.contains(quality)) {
                                    matchCount++;
                                }
                            }
                            
                            if (matchCount > 0) {
                                matchingBreeds.add(breedName + " (" + matchCount + " matches)");
                            }
                        }
                    }
                }
                
                if (matchingBreeds.isEmpty()) {
                    textArea.append("No breeds found matching the specified qualities.");
                } else {
                    textArea.append("Found " + matchingBreeds.size() + " matching breed(s):\n");
                    for (String breed : matchingBreeds) {
                        textArea.append("\nâ€¢ " + breed);
                    }
                }
                
            } catch (IOException e) {
                textArea.append("Error reading file: " + e.getMessage());
            }
        }
    }

    public static void main(String[] args) {
        new Web_res_lab();
    }
}
